-------------------

    >>> MyList = __import__('1-my_list').MyList

    >>> my_list = MyList()
    >>> my_list.append(1)
    >>> my_list.append(4)
    >>> my_list.append(2)
    >>> my_list.append(3)
    >>> my_list.append(5)

    [1, 4, 2, 3, 5]
